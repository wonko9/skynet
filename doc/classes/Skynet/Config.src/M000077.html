    def self.queue_name_by_id(queue_id)               
      queue_id = queue_id.to_i
      if Skynet::CONFIG[:MESSAGE_QUEUES][queue_id]
        return Skynet::CONFIG[:MESSAGE_QUEUES][queue_id]
      else     
        raise Skynet::Error("#{queue_id} is not a valid queue_id")
      end
    end
