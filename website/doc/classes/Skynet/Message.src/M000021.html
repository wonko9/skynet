    def self.new_task_message(task,job)
      self.new(
        :job_id       => job.job_id,
        :expire_time  => job.start_after,
        :version      => job.version,
        :queue_id     => job.queue_id || 0,
        :iteration    => 0,
        :tasktype     => :task, 
        :task_id      => task.task_id,
        :payload      => task,
        :payload_type => task.task_or_master,
        :expiry       => task.result_timeout, 
        :name         => task.name,       
        :retry        => task.retry
      )
    end
